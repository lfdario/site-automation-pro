{{/* Override sicuro per PaperMod: - Preferisce .Params.images[0] - Se non c'è, usa .Params.cover solo se è una stringa URL */}} {{- $cxt := or (.cxt) . -}} {{- $isHidden := or (.isHidden) false -}} {{- if not $isHidden -}} {{- $img := "" -}} {{/* 1) prova con .Params.images[0] */}} {{- with $cxt.Params.images -}} {{- if gt (len .) 0 -}} {{- $img = index . 0 -}} {{- end -}} {{- end -}} {{/* 2) fallback con .Params.cover se è stringa URL */}} {{- if not $img -}} {{- $cover := $cxt.Params.cover -}} {{- if $cover -}} {{- $coverStr := printf "%v" $cover -}} {{- if or (hasPrefix $coverStr "http://") (hasPrefix $coverStr "https://") -}} {{- $img = $coverStr -}} {{- end -}} {{- end -}} {{- end -}} {{- if $img -}} {{- end -}} {{- end -}}
